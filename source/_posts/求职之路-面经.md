```python
#!/usr/bin/python
# -*- coding: UTF-8 -*-
def max_sub_str(in_str):
   n = len(in_str)
   if n == 0:
      return 0
   res = 0 # 这个最小的值应该是1
   start = 0
   while start < n: # 这里终止条件应该写成n-1，开启的位置如果到了最后一个位置，没必要进行了。因为最小的字符长度就是1
      sub_str = ""
      for i in range(start, n):
         if in_str[i] not in sub_str:
            sub_str += in_str[i]
         else:
            res = max(res, len(sub_str))
            start = i # 这里应该写成start += 1
            break
   return res

s="abcabcbb"
res=max_sub_str(s)
print(res)
```
